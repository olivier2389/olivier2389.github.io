// ─── NORMALIZACIÓN CORREGIDA ────────────────────────────────── function normalizarHabilidades(data) { const r = []; if (data && data.cycles) { data.cycles.forEach(c => { const ciclo = c.ciclo || c.name || ''; (c.niveles || c.levels || []).forEach(n => { // ✅ CORREGIDO: extraer nivel correctamente const nivel = n.nivel || n.name || ''; (n.areas || []).forEach(a => { const area = a.nombre || a.name || ''; (a.habilidades || []).forEach(h => { if (h.id) { r.push({ id: h.id, texto: h.descripcion || h.habilidad || '', ciclo, area, nivel // ✅ ahora nivel se asigna correctamente }); } }); }); }); }); } console.log("📊 Habilidades normalizadas:", r.length); console.log("📊 Ejemplo primera habilidad:", r[0]); return r; } // ─── CONSTRUCCIÓN DE CICLOS_MAP CORREGIDA ──────────────────── function inicializarConData(data) { console.log("🔄 Inicializando con data..."); todasHabilidades = normalizarHabilidades(data); filtradas = [...todasHabilidades]; // Construir CICLOS_MAP CICLOS_MAP = {}; if (data && data.cycles) { const cicloLabels = { 'PRIMER_CICLO': 'Primer Ciclo (1°–3°)', 'SEGUNDO_CICLO': 'Segundo Ciclo (4°–6°)', 'TERCER_CICLO': 'Tercer Ciclo (7°–9°)', 'CICLO_DIVERSIFICADO': 'Ciclo Diversificado (10°–11°)', }; data.cycles.forEach(c => { const key = c.ciclo || c.name || ''; if (!key) return; const areasSet = new Set(); const aniosSet = new Set(); (c.niveles || c.levels || []).forEach(n => { const nivel = n.nivel || n.name || ''; if (nivel) aniosSet.add(nivel); (n.areas || []).forEach(a => { const nombre = a.nombre || a.name || ''; if (nombre) areasSet.add(nombre); }); }); CICLOS_MAP[key] = { label: cicloLabels[key] || key, areas: Array.from(areasSet), anios: Array.from(aniosSet).sort() }; }); } console.log("✅ CICLOS_MAP construido:", Object.keys(CICLOS_MAP)); console.log("✅ Áreas por ciclo:", Object.fromEntries( Object.entries(CICLOS_MAP).map(([k, v]) => [k, v.areas]) )); console.log("✅ Años por ciclo:", Object.fromEntries( Object.entries(CICLOS_MAP).map(([k, v]) => [k, v.anios]) )); poblarCiclos(); renderLista(filtradas.slice(0, 80)); document.getElementById('lista-info').textContent = todasHabilidades.length + ' habilidades · mostrando 80'; document.getElementById('loadingScreen').classList.add('hidden'); } // ─── POBLAR CICLOS CON ORDEN ───────────────────────────────── function poblarCiclos() { const sel = document.getElementById('sel_ciclo'); if (!sel) return; sel.innerHTML = ''; // Orden predefinido const orden = ['PRIMER_CICLO', 'SEGUNDO_CICLO', 'TERCER_CICLO', 'CICLO_DIVERSIFICADO']; orden.forEach(key => { if (CICLOS_MAP[key]) { sel.innerHTML += ``; } }); sel.disabled = false; console.log("✅ Ciclos poblados:", sel.options.length - 1); } // ─── ON CICLO CHANGE CORREGIDO ─────────────────────────────── function onCicloChange() { const ciclo = document.getElementById('sel_ciclo').value; console.log("🔄 Ciclo seleccionado:", ciclo); const aS = document.getElementById('sel_area'); const nS = document.getElementById('sel_nivel'); aS.innerHTML = ''; nS.innerHTML = ''; if (ciclo && CICLOS_MAP[ciclo]) { // Poblar áreas CICLOS_MAP[ciclo].areas.forEach(a => { aS.innerHTML += ``; }); // Poblar años CICLOS_MAP[ciclo].anios.forEach(a => { // Formatear año (1,2,3 → 1°,2°,3°) const anioFormateado = a.includes('°') ? a : `${a}° año`; nS.innerHTML += ``; }); } filtrarHabilidades(); } // ─── FILTRAR HABILIDADES CORREGIDO ─────────────────────────── function filtrarHabilidades() { const ciclo = document.getElementById('sel_ciclo').value; const area = document.getElementById('sel_area').value; const nivel = document.getElementById('sel_nivel').value; console.log("🔍 Filtrando:", {ciclo, area, nivel}); filtradas = todasHabilidades.filter(h => { const matchCiclo = !ciclo || h.ciclo === ciclo; // Normalizar áreas para comparación (sin tildes) const areaNorm = area ? area.toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, "") : ""; const hAreaNorm = h.area ? h.area.toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, "") : ""; const matchArea = !area || hAreaNorm === areaNorm; const matchNivel = !nivel || h.nivel === nivel; return matchCiclo && matchArea && matchNivel; }); console.log("✅ Filtrado:", filtradas.length, "habilidades"); renderLista(filtradas.slice(0, 80)); document.getElementById('lista-info').textContent = `${filtradas.length} habilidades · mostrando ${Math.min(filtradas.length, 80)}`; }
